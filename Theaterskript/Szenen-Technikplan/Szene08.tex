\section{Eine neue Welt\dots}
\DisplayPersons
\DisplayRequisites
\ort{In Schutztopia}

\StageDir{\lichtblitzd, \knalld. Licht an. \miladh, \alberth, \wilmah und \wilhelminah liegen auf dem Boden und rappeln sich auf.}

\licht{grün, orange, gelb}
\mikroan{1}
\mikroan{2}
\mikroan{3}
\mikroan{4}

\miladx{\direct{schreit} Aua, ging das nicht vielleicht ein bisschen vorsichtiger?}
\albertx{Was meckerst du schon wieder so? Ich kann da nichts für! Freu dich doch lieber, wir sind in einer neuen Welt!}
\wilmax{\direct{setzt sich auf} \textbf{Büüüüüügeleisen!!}}
\wilhelminax{\direct{wütend} Wir sind solche Nudeln!}
\wilmax{\direct{verzweifelt} Wieso sind wir jetzt mitgekommen?}
\wilhelminax{Wir wollten doch in unserer Welt bleiben!}
\wilmax{Verdammter Mistdreck, und das ganze hat uns jetzt zwei Wunschpunkte gekostet!}
\wilhelminax{Egal, wir wünschen uns einfach wieder zurück.}
\customx{\wilmah und \wilhelminah gemeinsam}{\direct{schließen die Augen} Ratzupaltuff!}

\StageDir{Kein Geräusch ertönt, kein Lichtblitz. \wilmah und \wilhelminah machen die Augen wieder auf, schauen sich verdutzt um.}

\wilmax{Aber nein, Wildschwein!}
\wilhelminax{\direct{verzweifelt} Warum funktioniert das nicht?!}
\albertx{\direct{belustigt} Ach, und du nennst dich Fluchkönigin des Jahres?! Ist doch kinderleicht! Ihr seid in einer neuen Welt, neue Gesetze, kein Wutpunktesystem mehr\dots\ Klingelt da was?}
\customx{\wilmah und \wilhelminah gemeinsam}{Neeeein! Das darf doch nicht \textbf{Warzenschwein}! Nudel, Nudel, \textbf{NUDEL!}}

\StageDir{\cosmah betritt die Bühne und beugt sich über die vier Ankömmlinge.}
\mikroan{5}

\cosmax{\direct{übertrieben freundlich} Hallo? Wer seid ihr denn? Alles in Ordnung? Braucht ihr Hilfe?}
\miladx{Jaja klar, wir müssen uns nur erstmal zurechtfinden. Wo sind wir hier gelandet? Können Sie uns helfen?}
\cosmax{\direct{fürsorglich} Aber natürlich, meine Liebe. Habt ihr euch verlaufen? Ich helfe euch gern, den Weg zu finden. Ich bin übrigens Cosma Fürsorglich!}
\wilmax{Ahh, das ist grauenhaft, viel zu freundlich!!}

\StageDir{\wilmah und \wilhelminah sind komplett schwarz gekleidet, während \cosmah ein buntes Kostüm trägt.}

\miladx{\direct{verwirrt} Ehm, ja, vielen Dank, Cosma! Das ist nett. Also ehrlich gesagt weiß ich gar nicht, wo wir hin wollen\dots}
\cosmax{Kein Problem! Das finden wir schon noch heraus! Ihr seid vollkommen verwirrt, oder? Wir befinden uns hier übrigens auf Schutz-Topia! Kommt doch erstmal mit zu mir und ruht euch etwas aus!}
\miladx{Ja, warum nicht?}
\albertx{\direct{nörgelnd} Och man, wie langweilig! Lasst uns doch lieber den Planeten entdecken!}
