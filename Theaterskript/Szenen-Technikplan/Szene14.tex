\section{Genug gechillt!}
\DisplayPersons
\DisplayRequisites

\ort{Irgendwo drinnen in der gechillten Welt.}

\StageDir{\miladh, \alberth, \wilmah, \wilhelminah und \cosmah sind wegen des Sturms reingegangen. Sie sitzen auf dem Boden und beraten sich.}

\miladx{Okay, Leute. Wir müssen hier weg! Diese Sorglosigkeit bereitet mir Sorgen!}
\cosmax{Ja, geht mir genauso.}
\wilmax{So ein Dreck! Ich will einfach nur wieder in meine Welt zurück!}
\wilhelminax{\direct{verzweifelt} Ja, unbedingt! Das ist schlecht für unsere Wutpunkte-Statistik, wenn wir so lange weg sind!}
\albertx{\direct{enttäuscht} Ach, kommt schon. Wir haben uns hier doch noch gar nicht richtig umgesehen!}
\miladx{\direct{genervt} Man Albert, lass' uns doch mal mit deinem blöden Entdeckerdrang in Ruhe!}
\wilmax{Genau!}
\wilhelminax{\textbf{Du Nudel!}}

\StageDir{\salomeh kommt auf die Bühne.}

\salomex{Wollt ihr nicht wieder rauskommen? Es ist doch eigentlich echt \textit{groovy} bei uns.}
\miladx{Nein, ich will einfach nur von hier weg!}
\cosmax{\direct{freudig} Zum Glück habe ich dafür vorgesorgt!}
\wilhelminax{\direct{verdreht die Augen} Was hast du denn jetzt schon wieder?}
\cosmax{Ich will ja nicht so enden wie ihr, und nicht mehr in meine Welt zurückkommen.}
\wilmax{\direct{unterbricht sie} Ach, sei doch still!}
\cosmax{\direct{ignoriert den Kommentar} Deshalb habe ich vorgesorgt, und meinen \kuchend eingepackt! Damit können wir überallhin reisen!}
\albertx{\direct{verächtlich} Jaja genau, das ist ja noch bescheuerter als deine Traumreise!}
\miladx{Aber beim letzten Mal hat es auch funktioniert!}
\wilmax{\direct{genervt} Ich hab einfach auf gar nichts mehr Bock.}
\wilhelminax{Eure blöden Spielchen gehen mir echt auf die Nerven!}
\wilmax{Genau!}
\customx{\wilmah und \wilhelminah gemeinsam}{Ihr \textbf{NUDELN!}}
\miladx{Ach kommt schon, ihr wollt doch auch wieder nach Hause. Lasst es uns einfach ausprobieren! Komm, \salomeh, du auch!}
\salomex{Ach, muss das sein? Ihr seid so crazy! Na gut\dots}

\StageDir{\cosmah verteilt den \kuchend, alle beißen hinein, \lichtblitzd und \knalld, Licht aus.}
