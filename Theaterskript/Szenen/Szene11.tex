\section{Liebes\dots}
\DisplayPersons
\DisplayRequisites

\ort{In \cosmahh 's Haus}

\StageDir{\wilmah, \wilhelminah, \miladh und \alberth sitzen am Tisch, \cosmah kommt später dazu.}

\begin{drama}
\wilmax{Man, ihr Nudeln, das ist ja echt grauenhaft hier!}
\wilhelminax{Wir müssen hier weg! Ich will nach Hause!}
\miladx{Ja, ich wäre auch gern woanders. Ich weiß ja nicht, was ihr heute so erlebt habt, aber ich fand's echt ätzend.}
\albertx{Ach, warum wollt ihr denn schon wieder gehen? Wir haben uns hier doch noch gar nicht richtig umgeschaut!}
\cosmax{\direct{kommt dazu} Wilma, Wilhelmina, Mila, ihr wollt weg?! Oh nein, hab ich irgendetwas falsch gemacht?}
\wilhelminax{Argh, du Nudel, hör bitte einfach auf, so freundlich zu sein!!}
\cosmax{Und kannst du bitte aufhören, so unfreundlich zu sein? Aber so nett, wie ich bin, helfe ich euch natürlich trotzdem gerne!}
\wilmax{Ach, und wie, bitteschön? Ihr habt hier ja kein Wunschpunktesystem!}
\wilhelminax{So ein Dreck.}
\cosmax{Ach, noch viel besser! Ich kann euch mit auf eine Traumreise nehmen! Dann könnt ihr mir gemütlich zuhören, entspannen, und schon seid ihr woanders!}
\albertx{\direct{lacht} Hahaha, genau, ich kann nicht mehr! Eine Traumreise! \direct{verächtlich} So ein Quatsch!}
\wilmax{Ich will meine Wutpunkte wiederhaben! Scheißverein!}
\wilhelminax{Nudel!}
\cosmax{\direct{atmet tief durch} Puhh, ganz ruhig. Einmal Liebe, zweimal Liebe, dreimal Liebe, viermal Liebe, fünfmal Liebe\dots}
\miladx{\direct{flüstert} Was macht sie da?}
\cosmax{Das hab ich gehört! Mich beruhigen, ihr macht mich \textbf{WAHNSINNIG!}}
\miladx{Lasst uns das doch wenigstens mit der Traumreise ausprobieren. Schaden kann es uns nicht, und ich will hier nicht weiter feststecken!}
\albertx{\direct{verächtlich} Okay, du kannst dich ja gerne selbst davon überzeugen, dass das nicht funktioniert.}
\cosmax{Los gehts!}
\end{drama}

\StageDir{Licht aus.}
