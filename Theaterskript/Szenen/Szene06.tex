\begin{drama}
\aladdinx{Pooh, wir wissen, dass du für die Rote Königin arbeitest. Erzähl uns ihren Plan!}
\winniex{\direct{schleckt schmatzend den letzten Honig von einem Löffel} Wieso?}
\jackx{\direct{wütend} Warum nicht?! Spuck's aus!}
\winniex{Und wieso?}
\aladdinx{\direct{lässig, cool} Sag es doch einfach.}
\schneewittchenx{Ihr macht das ja vollkommen falsch! \remy hätte andere mit dem Verhör beauftragen sollen, bei euch wird das doch nie was!}
\jackx{\direct{gekränkt} Und wie würde es Hoheit dann machen, wenn ich fragen darf?}
\schneewittchenx{Ich würde ihn mit Honig bestechen, aber wenn du es besser weißt, mach, wie du denkst!}
\jackx{\direct{verunsichert} Äh\dots\ Ja\dots\ Genau daran hatte ich auch gerade gedacht!}
\schneewittchenx{\direct{ungläubig} Natürlich.}
\aladdinx{Hört auf zu diskutieren, wir müssen die Welt retten!}
\jackx{Hast ja recht.}
\end{drama}

\StageDir{\schneewittchenh holt ein Glas Honig und geht damit zu \winnie, \jackh, \aladdinh und \pocahontash stellen sich um \winnie herum}

\begin{drama}
\schneewittchenx{Möchtest du ein leckeres Glas Honig?}
\winniex{\direct{kann nicht widerstehen} Jaaa, und wiee!}
\aladdinx{Schön. Dazu musst du uns aber den Plan der Königin verraten!}
\winniex{\direct{zögert}}
\schneewittchenx{Willst du den Honig, oder nicht? Sonst werde ich den Honig wohl selbst essen \direct{fängt langsam an, den Honig zu essen}}
\winniex{\direct{panisch} Ja ja, schon gut, ich erzähle euch alles!}
\end{drama}

\StageDir{\schneewittchenh gibt \winnieh den Honig, dieser reißt das Glas sofort an sich und umklammert es mit beiden Händen}

\begin{drama}
\jackx{Also?}
\winniex{\direct{es platzt aus ihm heraus} Die Rote Königin hat \willy Wonka erpresst, damit er Schokolade herstellt, die die Hypnose verstärkt! Sie hat mir versprochen, dass ich ein Zimmer voll mit Honig bekomme in ihrem Schloss, wenn ich sie unterstütze!}
\pocahontasx{Na, wer sagt's denn? Der Natur kann man einfach nicht widerstehen, sie ist einfach zu süß! \direct{Isst ein bisschen Honig}}
\winniex{Hey! Das ist ganz allein MEIN Honig! Meiner!}
\schneewittchenx{\direct{triumphierend} Seht ihr? Was habe ich euch gesagt? Man muss nur die richtigen Druckmittel kennen!}
\jackx{Ja, du bist ganz toll, \schneewittchen, wir wissen es.}
\aladdinx{Los, wir sollten schleunigst die anderen informieren. \pocahontas, pass' bitte auf unseren Gast auf, ja?}
\pocahontasx{Ich werde ihn nicht aus den Augen lassen!}
\end{drama}
