\section{Ich hasse Menschen!}
\DisplayPersons
\DisplayRequisites
\ort{Irgendwo in der Wutwelt}

\StageDir{\wilhelminah und \wilmah stürmen auf die Bühne, \alberth hinterher. Die Meute haben sie scheinbar abgeschüttelt.}

\begin{drama}
\wilmax{Verdammt, kannst du mal aufhören, uns zu verfolgen?! Ich hasse Menschen!}
\wilhelminax{Genau! Kümmer' dich doch um deinen eigenen Kram! Scheißverein!}
\albertx{Tut mir Leid, aber ich finde das alles so interessant. Besonders Ihren Fernsprecher! Habt ihr noch so technische Neuheiten?}
\wilmax{Neuheiten? Hast du hinter dem Mond gelebt oder was?}
\wilhelminax{Das ist doch längst nichts neues mehr! Hast du noch nie von einem Handy gehört, du}
\customx{\wilmah und \wilhelminah}{NUDEL?!}
\albertx{\direct{schaut verdutzt drein, schweigt}}
\wilhelminax{Wow, du kennst dich ja echt überhaupt nicht aus\dots}
\albertx{\direct{empört} Also, da wo ich herkomme\dots}
\wilmax{\direct{unterbricht \alberthh} Einen Moment bitte, ich habe noch etwas Wut in mir!}
\wilhelminax{Ich auch!}
\customx{\wilmah und \wilhelminah abwechselnd}{Aaaargh! Mist! Verdammt! Du Nudel! Hühnerkacke!}
\wilhelminax{Das ist hier ganz üblich auf diesem Planeten. Wunder' dich nicht!}
\albertx{Auf welchem wunderbaren, technisch weiterentwickelten Planeten befinden wir uns denn?}
\wilmax{\direct{entsetzt} Wo kommst du denn her? Hast du zu lang geschlafen, oder bist vor Wut vor die Tür gerannt? Also wir befinden uns hier in\dots}
\wilhelminax{\direct{stolz} Wut-Mania, der Planet, auf dem Hass Gottes Nächstenliebe ist!}
\albertx{\direct{begeistert} Wut-Mania, wie brilliant! Habt ihr denn irgendwelche besonderen Fähigkeiten? Könnt ihr euch beamen oder zaubern oder sowas?}
\wilhelminax{\direct{gleichgültig} Nur das übliche\dots}
\wilmax{Du weißt ja scheinbar gar nichts. Hier gibt es Wutpunkte, die man bekommt, wenn man genügend flucht und Hass verteilt. Und die kann man halt in Wunschpunkte umwandeln.}
\wilhelminax{Wunder' dich nicht, du sprichst hier nämlich mit den Rekordhaltern, momentan haben wir einen Punktestand von 10.998 Punkten! Da fällt mir ein:}
\customx{\wilmah und \wilhelmina zusammen}{Du NUUUDEL!!!}
\end{drama}

\StageDir{\plakatbd wird hochgehalten.}

\begin{drama}
\customx{\computerstimmed}{Herzlichen Glückwunsch! Ihr Wutpunktestand ist nun bei 10.999 Wutpunkten! Es fehlt nur noch ein Wutpunkt, bis ein neuer Wunschpunkt freigeschaltet wird! Verteilen Sie noch mehr Hass, dann knacken Sie vielleicht noch den Weltrekord!}
\albertx{\direct{verwundert} Wunschpunkte?}
\wilmax{Damit kann man sich jeden Wunsch efüllen!}
\wilhelminax{Siehst du das in unserem Gesicht?}
\albertx{\direct{beeindruckt} Wie beim Sams?! Das ist ja krass.}
\wilmax{Sams..?}
\albertx{Egal. Oh, darf ich mir bitte was wünschen?! Das wäre so cool! Zum Beispiel eine Maschine, mit der man komplizierte Mathematik in sekundenschnelle ausrechnen kann! Ja, genau, am besten noch für die Hosentasche zum mitnehmen!}
\customx{\wilmah und \wilhelminah zusammen}{Ehm\dots So etwas gibt es doch längst! \direct{\wilmah hält ihr Telefon hoch}}
\albertx{\direct{trotzig} Jaja, reinlegen kann ich mich selbst. Damit hast du doch noch gerade telefoniert! Tzz, dumm bin ich jetzt auch nicht.}
\wilhelminax{Nein, das kann beides! Und noch viel mehr!}
\albertx{\direct{verblüfft} Waas?! Ich bin beeindruckt!}
\end{drama}

\StageDir{\wilmah und \wilhelminah lachen.}

\begin{drama}
\wilhelminax{\direct{ironisch} Und du nennst dich Wissenschaftler?}
\albertx{\direct{bettelnd} Aber dann bitte bitte was anderes! Ich würde mir auch nur Gutes wünschen!}
\wilmax{\direct{angewidert} Nur Gutes?}
\wilhelminax{Das klingt ja grauenhaft!}
\albertx{\direct{korrigiert sich} Ehm, ich meinte natürlich nur Hass. Ich würde nur Hass verteilen! A propos Hass, wo ist eigentlich Mila?}
\wilmax{Was sollen wir denn wissen, wo du deine Freundin vergessen hast? Jede Sekunde, die wir mit dir verbringen, verschwenden wir Zeit!}
\wilhelminax{Ja, genau, wertvolle Zeit, die wir sonst zum Fluchen verwenden könnten!}
\albertx{Okay, Plan: Ihr wollt mich loswerden, und Mila will von hier weg. Dann wünscht uns doch zu Mila, und uns beide von diesem Planeten, dann habt ihr Ruhe!}
\wilmax{Was auch immer. Hauptsache, es geht schnell!}
\customx{\wilmah und \wilhelminah zusammen}{Ratzupaltuff!}
\end{drama}
