\Scene{Eine geglückte Mission}
\DisplayPersons
%\chars{Helge Schneider}
\ort{Im Geheimversteck}
\requ{Eine Banane}

\StageDir{\aladdinh, \remyh, \pocahontash und \meridah sitzen im Geheimversteck und spielen Karten. \winnieh ist nach wie vor auf dem Stuhl gefesselt und löffelt Honig. \jackh, \olafh, \doch, \schneewittchenh, \thomash und \willyh kommen auf die Bühne.}
\mikroan{1,2,3,4,5,6,8}

\begin{drama}
\meridax{\direct{schaut \willy an} Wer ist das denn?}
\willyx{Mein Name ist \willy, mir gehört die Schokoladenfabrik!}
\aladdinx{Aha! \direct{verschränkt die Arme} Ich dachte, wir nehmen keine Gefangenen?}
\meridax{\winnie ist doch auch einer, du Schlaumeier!}
\thomasx{\willyy ist gar kein Gefangener.}
\olafx{Genau, er hat uns gerettet! \direct{stolz}}
\schneewittchenx{Das wäre gar nicht nötig gewesen, wenn du uns gesagt hättest, dass du nicht pfeifen kannst!}
\jackx{Genau, und wenn Sie, Mylady, den Plan nicht vergessen hätten!}
\meridax{Das verstehe ich jetzt nicht. Was ist genau passiert?}
\thomasx{Also, es war so\dots}
\end{drama}

\StageDir{Alle setzen sich hin}

\begin{drama}
\thomasx{Also irgendwie war das doch noch nicht ganz so klar mit dem Plan, aber wir haben es trotzdem in die Fabrik geschafft.}
\schneewittchenx{Und meine Frisur ist total ruiniert! Und diese Inneneinrichtung\dots}
\thomasx{Ähh, ja. Auf jeden Fall wurden wir gefangen genommen, doch dann stellte sich heraus, dass \willyy auf unserer Seite steht, und gemeinsam haben wir ein Feuer gelegt und die Fabrik schleunigst verlassen.}
\docx{Genau! Und jetzt sind wir hier.}
\meridax{Na toll. \direct{schmollt} Und ich habe den ganzen Spaß wegen \aladdin verpasst!}
\aladdinx{Ist doch nicht meine Schuld, wenn du so\dots}
\remyx{Jetzt geht das schon wieder los. Freut euch doch einfach, dass die Mission geglückt ist! Vielen Dank für deine Hilfe, \willyy. Damit hätten wir wohl alle nicht gerechnet. \direct{Wendet sich an \willyy} Wenn du die Schokolade zubereitet hast, kannst du uns doch sicher auch was zu dem Gift sagen, oder?}
\willyx{Ja sicher. Im Prinzip ist es einfach nur eine Mischung aus verschiedensten Zutaten, darunter\dots}
\end{drama}

\mikroaus{1,2,3,4,5,6,8}
\StageDir{Während \willyh's letztem Satz: Licht aus.}
