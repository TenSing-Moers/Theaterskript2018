\section{Nichts wie weg!}
\DisplayPersons
\DisplayRequisites

\ort{Irgendwo in der Wutwelt}

\StageDir{\lichtblitzd, \knalld, Licht an. \alberth, \wilmah und \wilhelminah stehen bei \miladh auf der Bühne. \miladh rennt auf \alberth zu und umarmt ihn.}

\miladx{Oh Gott, Albert, da bist du ja!}
\albertx{\direct{drückt \miladh angeekelt weg} Ja, freut mich ebenfalls\dots\ Glaub' ich.}
\wilmax{Verflixte Axt.}
\wilhelminax{\direct{angewidert} Liebe, Igitigitt!}
\albertx{\direct{erinnert sich} Ach ja, stimmt ja, da war ja was. Mila, du, du, ehm\dots}
\wilmax{\direct{unterbricht ihn} Scheißverein, du kannst ja gar nichts!}
\wilhelminax{Mila, du NUDEL! SO macht man das!}
\miladx{Ehm, freut mich ebenfalls, dich kennenzulernen?}
\wilmax{\direct{arrogant} Ich bin Wilma Wutentbrannt, und die Person, die am besten Fluchen kann! Ernsthaft, ich bin sogar Jahresbeste!}
\wilhelminax{Und ich bin ihre Schwester. Zweitbeste Flucherin. Klingt gefährlich? Danke für das Kompliment!}
\miladx{Wutentbrannt? Warum wundert mich der beknackte Nachname jetzt nicht? \direct{zu \alberth} Deine Freundinnen sind mir sehr unsympathisch. Ich will einfach nur hier weg. An einen Ort, der nicht so voller Hass und Wut ist!}
\wilmax{\direct{ironisch} Freundinnen?}
\wilhelminax{Ha, da lache ich aber gleich los!}
\albertx{Wir sind keine Freunde. Ja, hast du denn eine Idee, wie wir hier wegkommen?}
\miladx{\direct{vorwurfsvoll} DU bist Schuld, dass wir überhaupt erst hier hergekommen sind. Also kannst DU dich auch um den Rückweg kümmern.}
\albertx{\direct{triumphiered} Na, dann kannst du mir ja dankbar sein, dass ich Wilma und Wilhelmine schon längst überredet habe, uns hier wegzuwünschen.}
\wilmax{Überredet? Wir sind einfach nur froh, euch loszuwerden.}
\wilhelminax{Genau, ihr nervt, ihr NUDELN!}
\albertx{\direct{verdreht die Augen} Soll ich dir jetzt auch noch die Füße küssen? Worauf warten wir noch!}
\customx{\wilmah und \wilhelminah gemeinsam}{\direct{schließen die Augen} Ratzupaltuff!}

\StageDir{\lichtblitzd, \knalld, Licht aus.}
