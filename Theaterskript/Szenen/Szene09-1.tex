\Scene{Vor der Schokoladenfabrik}
\DisplayPersons
%\chars{Helge Schneider}
\ort{Mitsibitsi-Elektro-Halle}
\requ{Pinke Sonnenbrillen, Sound eines Lieferwagens}

\StageDir{\thomash, \olafh, \jackh, \doch und \schneewittchenh stehen in Sichtweite der Schokoladenfabrik, tragen pinke Sonnenbrillen und beraten sich über das weitere Vorgehen}
\mikroan{1,2,3,4,5}

\begin{drama}
\thomasx{Okay, haltet euch an den Plan, los geht's!}
\jackx{Äh\dots\ Ja, genau\dots\ Der Plan\dots}
\olafx{Ehm, was war denn nochmal der Plan?}
\schneewittchenx{Hä? Habt ihr \remy nicht zugehört oder was?}
\docx{Ja genau, habt ihr nicht zugehört?}
\end{drama}

\StageDir{\olafh, \jackh und \thomash drucksen herum}

\begin{drama}
\olafx{Also\dots\ schon\dots\ aber\dots}
\jackx{Ich höre niemals bei Plänen zu, die funktionieren sowieso nicht!}
\thomasx{Also\dots\ Ich glaube, der Plan war\dots\ \direct{überlegt} Die Schokoladenfabrik zu zerstören! Am besten, wir gehen erstmal rein!}
\olafx{Au ja, super Idee!}
\jackx{Und wie gedenkst du, das anzustellen?}
\docx{Ich habe natürlich zugehört! Und der Plan war\dots\ ehm\dots}
\jackx{Ja? Was war der Plan?}
\docx{Ehm\dots\ Ich hab's gleich\dots\ \direct{überlegt}}
\end{drama}

\sound{Lieferwagen}
\StageDir{Ein Lieferwagen nähert sich dem Lieferanteneingang}

\begin{drama}
\docx{\direct{fängt sich wieder} Wir springen einfach blitzschnell auf die Ladefläche des Lieferwagens, der dort hinten um die Ecke kommt!}
\schneewittchenx{Spinnst du? Das ist viel zu gefährlich! Da brechen mir hinterher noch die Nägel ab!}
\olafx{Ach was, was soll da schon schiefgehen? \direct{will loslaufen}}
\thomasx{\direct{hält \olaf zurück} Also du bleibst erstmal schön hier und hältst Wache!}
\jackx{Genau, das ist ein seeehr wichtiger Job! Pfeif' einfach, wenn du jemanden kommen siehst, okay?}
\schneewittchenx{Hallo?! Hat mir überhaupt jemand zugehört? Ich springe nicht auf diesen Wagen, da brauchen wir auch gar nicht zu diskutieren!}
\thomasx{Stimmt, das ist wirklich riskant\dots\ Wir werden mit Sicherheit gesehen!}
\docx{Aber haben wir eine Wahl? Los jetzt, bevor es zu spät ist!}
\end{drama}

\StageDir{\schneewittchenh, \jackh, \thomash und \doch laufen los, Licht aus. Nach einer kurzen Pause (Applaus abwarten?) Licht wieder an}

\begin{drama}
\olafx{\direct{sieht den anderen nach} Aber\dots\ Ich kann doch gar nicht pfeifen! \direct{Hält sich alle 6 Finger vor den Mund und macht ein Furzgeräusch}}
\end{drama}

\mikroaus{1,2,3,4,5}
\StageDir{Licht aus.}
