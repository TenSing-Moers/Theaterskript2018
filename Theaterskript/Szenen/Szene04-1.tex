\Scene{Was ist hier eigentlich los?}
\DisplayPersons
%\chars{\pocahontash \delim \remyh \delim \schneewittchenh \delim \thomash}
\ort{Vor dem Geheimversteck der Guten}
\requ{Blumentopf, andere Blumen}

\StageDir{\remyh, \thomash, \schneewittchenh und \doch kommen auf die Bühne. \pocahontash läuft umher, hat einen Blumentopf in der Hand. Sie bemerkt die vier zunächst nicht.}
\mikroan{1,2,4,6,7}

\begin{drama}
\pocahontasx{\direct{Redet zu der Blume in ihrer Hand} Guten Morgen, meine Liebe! Hast du gut geschlafen? Ich fühle die Harmonie in der Umgebung!}
\thomasx{Ähh\dots\ Wer ist das denn?}
\schneewittchenx{Ach, das ist \pocahontas, sie ist ein bisschen seltsam\dots\ Redet ständig mit Pflanzen und Tieren und faselt was von Harmonie und Gleichgewicht\dots}
\pocahontasx{\direct{Bemerkt die Ankömmlinge} Hallo, wollt ihr zu den anderen? Moment\dots\ Wen habt ihr denn da dabei? Aber zuerst das Passwort!}
\remyx{Ach \pocahontas, das hatten wir doch schonmal\dots\ Du kennst uns doch!}
\pocahontasx{Ich weiß, aber das ist zu unser aller Sicherheit! Also. Das Passwort, bitte?}
\schneewittchenx{Meine Güte, na gut. Schwachkopf, Schwabbelspeck, Krimskrams, Quiek! Zufrieden? \direct{Wendet sich zu \thomas} Ich hab echt keine Ahnung, wie die darauf gekommen ist\dots}
\pocahontasx{Also gut, ihr könnt durch. Aber ihr habt mir immer noch nicht gesagt, wer das ist! Ist der überhaupt sicher? Darf ich ihn überhaupt vorbeilassen? Und seid ihr auch sicher, dass dieser neue die Harmonie dieses Ortes nicht zerstört? Ich habe lange daran gearbeitet, dass meine Pflanzen diesen Ort so lieben. Mit eurer Rücksichtslosigkeit habt ihr mir das ganz schön schwer gemacht!}
\thomasx{Also\dots\ Ich\dots}
\remyx{Jaja, schon klar, wir achten darauf. Wir waren halt bei dieser Castingshow und haben uns da mal umgesehen und so, und dann ist plötzlich \mickey auf die Bühne gekommen, hat alle bedroht und so. \thomas, so heißt dieser neue, wusste irgendwie nicht, wo er war, und war sehr verwirrt, also haben wir ihn erstmal mitgenommen. Später meinte er, dass er irgendwie aus einem anderen Land käme und hier hergebeamt wurde, aber\dots\ So ganz haben wir das wohl alle noch nicht verstanden. Wir fragen mal die anderen, und erzählen ihnen alles. Passt du weiter auf?}
\pocahontasx{Klar, niemand kommt hier ohne meine Erlaubnis vorbei! \direct{Wuselt von der Bühne, im Gehen:} Oh, guten Morgen, meine Vögelchen! Wie geht es euch? Was habt ihr mir zu erzählen?}
\schneewittchenx{\direct{Kopfschüttelnd} Ich sag doch, sie ist seltsam\dots}
\thomasx{Äh ja, ich merke schon\dots}
\docx{Los, wir sollten die anderen nicht länger warten lassen!}
\end{drama}
