\ort{Disneyland}
\requ{Stühle für den Moderator und den Kandidaten}

\StageDir{\thomash taucht unbemerkt in einer Gruppe von \customh{Kandidaten} am Bühnenrand auf, guckt sich verwirrt um. In der hinteren Bühnenmitte sind zwei Stühle gegenüber voneinander aufgestellt, auf einem sitzt der \moderatorh.}
\mikroan{1,5}

\begin{drama}
\thomasx{Äh\dots\ Was\dots\ Wo bin ich?}
\moderatorx{Der nächste Kandidat, bitte!}
\end{drama}

\StageDir{\remyh tritt aus der Menge hervor und setzt sich zum Moderator. \thomash entfernt sich ebenfalls von der Gruppe und läuft verwundert über die Bühne.}
\mikroan{6}

\begin{drama}
\moderatorx{Was sind deine Kompetenzen für deinen Job als Repräsentant von Disney?}
\remyx{Äh\dots\ Ich\dots\ kann kochen! Und viele Leute kennen mich!}
\moderatorx{Tut mir Leid, aber wer Disney repräsentieren will, muss allumfassend sein. Disney ist in den letzten Jahren groß geworden, wir brauchen etwas\dots\ besonderes! Du bist zu gewöhnlich und zu unbekannt.}
\end{drama}

\StageDir{\remyh verlässt den Moderator. \thomash läuft weiterhin ziellos auf der Bühne umher. \remyh erblickt Thomas und läuft zu ihm. \thomash und \remyh stehen am vorderen Bühnenrand, das Casting findet weiterhin in der hinteren Bühnenmitte statt. \customh{Spot} auf \thomas und \remy.}

\begin{drama}
\remyx{He, was machst du denn hier?}
\thomasx{Ich\dots\ Weiß nicht\dots\ Wo bin ich denn hier gelandet?}
\remyx{Mitten in der Castingshow! Hast du in den letzten Wochen auf dem Mond gelebt, oder wieso weißt du von nichts?}
\thomasx{Äh\dots\ Vermutlich\dots\ Wer oder was wird denn hier gecastet?}
\remyx{Na, das neue \glqq Aushängeschild\grqq\ von Disney! Du hast es doch soeben gehört!}
\thomasx{Achso\dots\ Ja\dots\ klar... Disney?}
\remyx{Ja was denn sonst? Schließlich sind wir hier in Disneyland!}
\thomasx{Disneyland?}
\remyx{Ja, Disneyland! Wiederholst du immer alles, was andere sagen?}
\thomasx{Nein, nur\dots\ Eben war ich noch\dots\ \direct{dreht sich verwundert um, verharrt kurz, und dreht sich wieder zurück} Warte mal... Du bist eine Ratte?! Und kannst reden?!}
\end{drama}

\StageDir{\customh{Spot} wechselt zum \moderator}

\begin{drama}
\moderatorx{\direct{gelangweilt} Der Nächste!}
\end{drama}

\StageDir{\schneewittchenh kommt mit \doch auf die Bühne.}
\mikroan{2}

\begin{drama}
\moderatorx{Was soll das denn? \direct{fasst sich an den Kopf} Nur eine Person, bitte! \direct{stöhnt}}
\schneewittchenx{\direct{zickig} Aber wir wollen nur zusammen als Symbol von Disney abgebildet werden!}
\end{drama}

\StageDir{\customh{Spot} wieder auf Thomas und Remy}

\begin{drama}
\remyx{Also, nochmal. Du weißt wirklich nicht, wo du bist?}
\thomasx{Nein, ich\dots\ Ich habe gar keine Ahnung. Also doch, vorhin war ich noch in Moers in mitten meiner eigenen Sendung, aber dann\dots}
\end{drama}

\StageDir{\thomash wird von \schneewittchenh und \doch unterbrochen, die sich zwischen \remyh und \thomash stellen. Beide sehen \remyh an, erblicken \thomash scheinbar noch gar nicht.}

\begin{drama}
\schneewittchenx{Ach, die wollen uns doch eh nicht. Sollen wir nicht lieber gehen? Wir haben doch sowieso schon alles gesehen hier\dots\ \direct{Erblickt Thomas, verwundert} Moment, was ist das denn für ein Typ?}
\end{drama}

\sound{Knall}
\StageDir{\remyh will zur Erklärung ansetzen, doch er kommt nicht dazu. \mickeyh stürmt (blutverschmiert?) auf die Bühne, ein Knall, wilde Farben, dazu Strobo}
\mikroan{3}

\begin{drama}
\schneewittchenx{Ach du meine Güte, was macht der denn plötzlich hier?}
\mickeyx{\direct{Zielt mit Waffe in die Runde} Stopp! Anhalten! Keiner bewegt sich, alles hört auf mein Kommando! Los, seht mich alle an!}
\end{drama}

\StageDir{\customh{Kandidaten} kreischen auf, schlagen die Hände über dem Kopf oder vor dem Mund zusammen etc., einige rennen verschreckt los, andere fallen in Ohnmacht}

\begin{drama}
\remyx{Was auch immer hier los ist, wir müssen verschwinden\dots\ Sofort!}
\end{drama}
\mikroaus{1,2,5,6}

\StageDir{Die vier zögern nicht lange. \schneewittchenh packt \thomash am Arm und zieht ihn hinter sich her, \remyh und \doch laufen den beiden voran von der Bühne. Nach und nach sehen alle zu \mickeyh, sobald sie diese erblicken, verfallen sie in Trance und verstummen.}

\begin{drama}
\mickeyx{Ruhe! Sofort! Schaut bitte alle zu mir!}
\end{drama}

\mikroan{4}

\begin{drama}
