\section{Guten Morgen!}
\DisplayPersons
\DisplayRequisites
\ort{In \miladhh 's Zimmer}

\StageDir{\miladh wird von \mutterh wachgerüttelt.}

\begin{drama}
\mutterx{Mila, wach jetzt auf! Es ist schon halb neun! Du hast schon verschlafen!}
\miladx{\direct{brummt verschlafen und genervt etwas unverständliches}}
\mutterx{Was hast du gesagt?! Geht es dir nicht gut? Bist du etwa heiser??! Oder hast du Halsschmerzen?}
\miladx{Boah nein Mama, nerv nicht! Ich bin einfach nur müde.}
\mutterx{Hast du gestern etwa wieder zu lange Tagebuch geschrieben? Du weißt doch, dass das deinem Immunsystem nicht gut tut! Naja, komm jetzt, zieh dich an, damit du noch vor dem Frühstück dein Zimmer aufräumen kannst!}
\end{drama}

\StageDir{\miladh stöhnt, und \mutterh geht von der Bühne. \leonh und \lenah kommen ins Zimmer.}

\begin{drama}
\leonx{Komm \miladh, wir spielen Piraten! Du bist die gefangene Prinzessin, und \lenah und ich kommen dich retten! Oder\dots Oder wir haben dich gefangen genommen! \direct{knurrt knurrt und wirft sich auf \miladh}.}
\lenax{Jaa! Bitte \miladh! Lass uns spielen!}
\miladx{\direct{super genervt, stöhnt} Zieht Leine, ihr kleinen Giftzwerge!}
\lenax{\direct{im Gehen} Boah, du bist so doof und langweilig, genau wie Mama!}
\leonx{Da hat sie Recht!}
\end{drama}

\StageDir{\leonh und \lenah haben die Bühne verlassen.}

\begin{drama}
\miladx{\direct{zu sich} Oh man, als ob ich genauso schlimm wäre wie Mama, die ständig fragt, ob es mir gut geht, und mir tausend Dinge aufträgt, die ich machen soll. Und andauernd denkt sie, ich wäre krank! Ich wünschte, ich wäre irgendwo, wo sich nicht alle ständig Sorgen umeinander machen, sondern wo einfach jeder sein Ding macht. Ein Ort, wo nicht ständig irgendwer was von mir will. Naja, die Träumerei bringt ja sowieso nichts, ich sollte mich für unsere Aufführung fertig machen.}
\end{drama}

\StageDir{Licht aus.}
