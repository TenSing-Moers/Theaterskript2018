\section{Der Überfall}
\DisplayPersons
%\chars{Helge Schneider}
\ort{In der Schokoladenfabrik}
\requ{Handspiegel, Pappwand zum Dahinter-verstecken o.ä., Fesseln für die Hände, Pistolen für die Wachmänner}

\StageDir{\thomash, \jackh, \schneewittchenh und \doch stehen in der Schokoladenfabrik und beraten über das weitere Vorgehen}
\mikroan{1,2,4,5}

\begin{drama}
\thomasx{So, jetzt sind wir drin. Und wie geht es jetzt weiter?}
\schneewittchenx{\direct{Holt einen Handspiegel heraus und begutachtet ihre Frisur von allen Seiten} Meine schöne Frisur ist vom Fahrtwind komplett ruiniert!}
\jackx{Frauen! Dafür ist jetzt wirklich keine Zeit! Wir müssen\dots}
\docx{Pscht! Ich glaube, ich höre da was!}
\end{drama}

\StageDir{Man hört \wacheah und \wachebh auf die Bühne kommen, sowie Furzgeräusche von \olafh, der immer noch versucht, zu Pfeifen. Die Wachen haben \olafh mit einem Seil die Hände verbunden und schubsen ihn vor sich her, halten ihn mit Pistolen unter Kontrolle}
\mikroan{3,6}

\begin{drama}
\docx{Schnell, versteckt euch!}
\schneewittchenx{Oh Gott! Ist das etwa \olaf?}
\jackx{Keine Bange, Mylady! Ich rette ihn!}
\thomasx{\direct{zischt} Nein, spinnst du?!}
\end{drama}

\StageDir{Doch da ist \jackh schon aus dem Versteck hervorgetreten und stellt sich den Wachen in den Weg}

\begin{drama}
\jackx{Mein Name ist Jack Sparrow, Capt'n Jack Sparrow, wenn ich bitten darf, und ich erwünsche die Aushändigung des Gefangenen!}
\end{drama}

\StageDir{\wacheah und \wachebh schauen sich kurz an und brechen in schallendes Gelächter aus}

\begin{drama}
\wacheax{Bitte was? Den Gefangenen willst du?}
\wachebx{Davon träumst du wohl. So, herkommen, dich nehmen wir auch noch mit.}
\end{drama}

\StageDir{\wacheah hält \olafh und winkt mit der Pistole \jackh zu \wachebh, \jackh streckt widerwillig die Arme aus, \wacheah fesselt sie.}

\begin{drama}
\schneewittchenx{Wir müssen doch etwas\dots}
\docx{\direct{Hält Schneewittchen den Mund zu} Psssch!}
\schneewittchenx{\direct{Fällt vor Schreck der Handspiegel herunter}}
\wacheax{Aha! Du hast also gleich noch deine Freunde mitgebracht?}
\wachebx{Na los, raus da! Ihr seid geliefert!}
\end{drama}

\StageDir{\wacheah und \wachebh winken die anderen mit den Waffen aus ihrem Versteck, \schneewittchenh, \thomash und \doch kommen mit erhobenen Händen hervor}

\begin{drama}
\wacheax{Na sieh mal einer an, wer seid ihr denn?}
\wachebx{Steckt ihr etwa alle unter einer Decke?}
\thomasx{Das ist alles ein großes Missverständnis! Wir wollten nur\dots}
\wacheax{Ist uns egal, was ihr vorhattet! Wir bringen euch erstmal allesamt zu\dots}
\end{drama}

\StageDir{\willyh kommt auf die Bühne.}

\begin{drama}
\willyx{Was ist hier los?}
\customx{\wacheah und \wachebh}{\direct{Im Chor} Hallo Chef! \direct{Heben die flache Hand an die Stirn zum Gruß}}
\wachebx{Diese Eindringlinge haben sich unbefugt Zugang zum Fabrikgelände verschafft, Chef!}
\wacheax{Wir waren gerade auf dem Weg zu ihnen, Chef!}
\willyx{Da habt ihr große Arbeit geleistet!}
\customx{\wacheah und \wachebh}{\direct{Im Chor} Danke, Chef!}
\willyx{Aber ab jetzt übernehme ich die Gefangenen. Wir gehen erstmal gemeinsam in mein Büro.}
\customx{\wacheah und \wachebh}{\direct{Im Chor} In Ordnung, Chef!}
\end{drama}

\StageDir{\wacheah und \wachebh händigen \willyh die Gefangenen aus und verlassen die Bühne}
\mikroaus{3,6}

\begin{drama}
\willyx{So, und ihr kommt jetzt erstmal mit. Na los, wird's bald?}
\end{drama}

\StageDir{\willyh scheucht die fünf vor sich her über die Bühne auf dem Weg zum Abgang}
\mikroaus{1,2,4,5}


\StageDir{Alle sechs verlassen die Bühne. Licht aus.}
