\section{Danke, Danke!}
\DisplayPersons
\DisplayRequisites

\ort{In \cosmahh 's Haus.}

\StageDir{Spot auf \milamh an der Bühnenkante.}

\begin{drama}
\milamx{Liebes Tagebuch, am Anfang war ich ja eigentlich ganz angetan von der neuen Situation. Die Hilfsbereitschaft der Leute hat mir gefallen, ganz anders als in der Hasswelt! Auch wenn Wilma, Wilhelmina und Albert mitgekommen sind, und ständig nerven\dots\ Fast wie meine Geschwister! Doch mit der Zeit hat mich die ganze Fürsorge einfach nur noch überfordert. Wir waren noch gar nicht richtig in Cosmas Wohnung angekommen, dann kamen auch schon die Nachbarn! Eigentlich ja ganz nett, aber sie wollten sich alle um mich kümmern, brachten merkwürdige Kleidung, Essen und sogar Medizin! Als ob es mir so schlecht ginge\dots}
\end{drama}

\StageDir{Auf der Bühne stehen mehrere Personen um \miladh herum, und reden auf sie ein.}

\begin{drama}
\charlottex{Hier noch ein paar \keksed! Die helfen dir bestimmt schnell wieder auf den Damm!}
\sophiax{Genau, gute Idee! Aber dann iss auch noch eine \bananed dazu, damit du genügend Vitamine hast!}
\cosmax{Und das Trinken nicht vergessen, hier ist \orangensaftd!}
\emmax{Ich habe \medizind dabei, vorbeugend gegen alle Krankheiten! Und zur Sicherheit habe ich auch noch \vbzeugd und Pflaster dabei!}
\cleox{Und hier hast du noch etwas \kleidungd, deine ist ja bestimmt ganz schmutzig, und wir wollen nicht, dass du dir von dem ganzen Dreck noch eine Krankheit holst!}
\miladx{Ähh\dots\ Ja, vielen Dank\dots\ Ich glaube, ich gehe mal kurz auf die Toilette\dots}
\cosmax{Oh ja, sehr gerne, ich begleite dich natürlich, und zeige dir alles!}
\end{drama}

\StageDir{\miladh und \cosmah stehen auf, alle anderen folgen ihnen. \miladh verlässt die Bühne, alle anderen warten an der Bühnenkante}

\begin{drama}
\charlottex{Kann man dir noch helfen? Brauchst du noch etwas \klopapierd? \direct{Hält \klopapierd hoch}}
\emmax{Ich habe auch noch ein paar \feuchttuecherd! \direct{Hält \feuchttuecherd hoch}}
\sophiax{Hier, versuch nicht, die Klobrille zu desinfizieren! \direct{Hält \desinfektionsmitteld hoch}}
\miladx{\direct{Schaut nochmal zwischen dem Bühnenvorhang hervor, nimmt widerwillig die Gegenstände an} Jaa, klar, natürlich!}
\cleox{Trink am besten noch etwas, damit es sich auch richtig lohnt! \direct{reicht \orangensaftd an}}
\end{drama}

\StageDir{\miladh nimmt auch noch den \orangensaftd an, ist total überfordert, verschluckt sich, die anderen Sachen fallen ihr herunter, alle reagieren total hysterisch und legen sie auf den Boden und reden wild durcheinander. Spot auf \milamh an der Bühnenkante:}

\begin{drama}
\milamx{Die ganze Situation hat mich total überfordert, als sie mich dann hingelegt hatten, war ich auch total erschöpft. Auch das war irgendwie nicht das richtige, ich wollte mal wieder nur weg. Warum kann es nicht einfach einen Ort geben, wo einem nicht alle auf die Nerven gehen?}
\end{drama}

\StageDir{Licht aus.}
