\section{Tagebuch\dots}
\DisplayPersons
\DisplayRequisites
\ort{Am Strand in der entspannten Welt}

\StageDir{\miladh wacht alleine am Strand auf, liegt auf dem Boden. \strandhandtuchd, \meeresrauschend. \salomeh sitzt neben \miladh, etwas abseits sitzen die anderen (\sophiah, wer noch?)}

\begin{drama}
\charlottex{Hey! Auch mal wach?}
\miladx{\direct{verschlafen, setzt sich auf} Hä?! Was?!}
\charlottex{Man. Du hast es aber drauf. So lange kann ich nicht pennen. Ich bin neidisch!}
\end{drama}

\StageDir{\miladh wird langsam wach, setzt sich auf und sieht sich um. Andere Menschen sitzen / liegen am Strand, achten nicht auf \miladh.}

\begin{drama}
\miladx{\direct{verwirrt} Wo bin ich denn jetzt schon wieder?}
\salomex{Na, hier bei uns! Chill' dich doch einfach dazu!}
\miladx{\direct{überrascht} Echt? Das ist aber nett, wie heißt du?}
\salomex{Ja klar doch! Ich bin Salome Sorglos!}
\miladx{Cool! Freut mich, dich kennenzulernen!}
\miladx{\direct{zu sich selbst} Man, endlich mal keine nervigen Fragen als Begrüßung. So könnte es immer sein!}
\miladx{\direct{Setzt sich mit \salomeh zu den anderen} Man, das ist ja echt chillig hier bei euch!}
\sophiax{Ist doch logisch!}
\end{drama}

\StageDir{\alberth, \wilmah, \wilhelminah und \cosmah kommen von der Seite dazu.}

\begin{drama}
\miladx{Oh man, die hatte ich ja schon ganz vergessen\dots}
\albertx{Mila, da bist du ja!}
\wilmax{\direct{ironisch} Wuhu! Ein weiteres Mal auf einer anderen, beknackten Welt\dots}
\wilhelminax{Scheißverein!}
\salomex{Also, ich finde unser Fleckchen hier ganz schön relaxt!}
\cosmax{\direct{fröhlich} Ach wunderbar, wie schön!}
\miladx{Seht ihr denn nicht, dass ich mich gerade mit Salome unterhalte? Und dir, Albert, kann ich nur sagen: Wie man sieht, hat die Traumreise sehr wohl funktioniert!}
\albertx{\direct{will etwas erwidern} Aber\dots}
\miladx{\direct{streckt \alberth abweisend die Hand entgegen und wendet sich wieder an \salomeh} Was macht ihr hier denn so den ganzen Tag?}
\salomex{Och, mal sitzen wir zu Hause, mal im Einkaufszentrum, aber die meiste Zeit sind wir hier am Strand.}
\miladx{Das klingt cool!}
\emmax{Schaut mal, ein Pikachu! \direct{zeigt irgendwo hin}}
\albertx{Aha, interessant, Pikachu? Was ist das?}
\wilhelminax{Ach Albert.}
\wilmax{Sei doch nicht immer so neugierig.}
\end{drama}

\StageDir{\pikachuh kommt auf die Bühne, \lospikachud aus dem Off.}

\begin{drama}
\sophiax{Pikachu? Witzig! Den habe ich ja seit Ewigkeiten nicht mehr gesehen!}
\miladx{Pikachu? Wo?}
\emmax{Na, da vorne!}
\albertx{Wo denn? Ich seh nichts!}
\end{drama}

\StageDir{\pikachuh verlässt die Bühne wieder.}

\begin{drama}
\miladx{Hä, wovon redet ihr?}
\salomex{Na, von Pikachu! Der ist doch da gerade langgelaufen mit einer neuen Sonnenbrille!}

\end{drama}
