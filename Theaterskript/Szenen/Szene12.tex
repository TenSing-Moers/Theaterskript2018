\StageDir{\mickeyh geht alleine im Zimmer auf und ab und führt Selbstgespräche, als hätte sie eine geteilte Persönlichkeit}
\mikroan{3}

\begin{drama}
\mickeyx{\direct{böse} Es ist wirklich höchste Zeit, dass ich alleine wieder die Herrschaft übernehme! \direct{verängstigt, traurig} Aber eigentlich will ich doch nur, dass mich alle wieder mögen! Was hat mir die \rk da bloß eingeredet? \direct{böse} Ach was, die \rk hat mir nur die richtige Idee gegeben, ich bin auf dem besten Weg!}
\end{drama}

\StageDir{\winnieh kommt auf die Bühne gestürmt}
\mikroan{6}

\begin{drama}
\winniex{Herr Mouse, ich habe dringlichste Neuigkeiten!}
\mickeyx{\direct{liebevoll} Ach \winniee, es ist so schön, dich zu sehen! \direct{böse} Ähm, ich meine, wo warst du denn so lange?!}
\winniex{Ich, Sir, wurde entführt!}
\mickeyx{Entführt? Von wem?}
\winniex{Also\dots\ Eine hatte rote Haare, einer einen langen Schwanz\dots\ Und dann war da noch einer, der war sogar noch kleiner als ich, und hatte eine schöne Mütze auf! Und so eine komische, die die ganze Zeit meinen Honig essen wollte\dots\ Und\dots}
\mickeyx{\direct{forsch} Hast du denn keine Namen?!}
\winniex{\direct{untergeben, duckt sich} Nein, Sir, leider nicht, aber ich glaube, es waren diese, ähm, wie nannten Sie sie nochmal\dots\ Rebellen?}
\mickeyx{Die Rebellen?! Warum sagst du das nicht gleich? Hast du etwas belauschen können?}
\winniex{\direct{untergeben} Belauscht? Also\dots\ Geredet haben die viel, und ich war mit meinem Honig beschäftigt\dots\ \direct{Ihm geht ein Licht auf} Aber doch, an eines erinnere ich mich! Ich glaube, sie wollen etwas kochen, Sir, gegen dieses\dots\ diese\dots\ dieses schwierige Wort, Sir, Hypothese?}
\mickeyx{\direct{entsetzt} Sie kochen ein Mittel gegen die Hypnose?!}
\mickeyx{\direct{untergeben} Ja genau, Sir, Sie haben wie immer Recht.}
\end{drama}

\StageDir{\rkh kommt mit großen Schritten auf die Bühne.}
\mikroan{4}

\begin{drama}
\rkx{Was höre ich da? Ein Gegengift?! Worauf wartest du noch, \mickey, wenn sie wirklich ein Mittel gegen die Hypnose gefunden haben, dann ist alles verloren! Ab mit ihren Köpfen! Los, \mickey, wir dürfen keine Zeit verlieren!}
\mickeyx{Ähm\dots\ Natürlich\dots\ Äh\dots\ \winniee?}
\winniex{Ja, Sir?}
\mickeyx{Zeig mir den Weg zu ihrem Versteck, sofort!}
\rkx{Sehr gut, \mickey, finde sie!}
\end{drama}
