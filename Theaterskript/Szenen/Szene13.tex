\section{Liebes Tagebuch\dots}
\DisplayPersons
\DisplayRequisites
\ort{Am Strand in der entspannten Welt}

\StageDir{Spot auf \milamh am Bühnenrand.}

\begin{drama}
\milamx{Liebes Tagebuch, heute war wieder ein ganz verrückter Tag. Irgendwie sind wir durch eine Traumreise in eine andere Welt gekommen! Ein schönes Gefühl, weil Albert erst nicht glauben wollte, dass das funktioniert. Diesmal hatte ich recht! Dann habe ich diese coolen, gemütlichen Menschen getroffen. Die waren echt nett, total entspannt! Irgendwie beeindruckend. Aber ich weiß nicht\dots\ Nach einiger Zeit ist mir bewusst geworden, \textbf{WIE} gleichgültig sie waren. Und irgendwie wirkten die ein bisschen ZU entspannt. Und wollten mir die ganze Zeit etwas andrehen, ich weiß gar nicht, was das war. Aber Geheuer war es mir nicht\dots}
\sophiax{\direct{zu \miladh} Klasse! Die Sonne geht schon langsam unter! Dann ist es endlich Zeit für's Abendessen!}
\miladx{Abendessen? Klasse! Und wie geht's danach weiter?}
\sophiax{\direct{gleichgültig} Och, ich denke mal, weiter chillen?}
\emmax{\direct{ergänzt} Und irgendwann mal schlafen gehen!}
\cosmax{Ernsthaft? Wie langweilig! Macht ihr denn den ganzen Tag nichts anderes?}
\cleox{Och, eigentlich nicht. Meistens schlafen wir auch bis in den Nachmittag\dots}
\cosmax{Bis in den Nachmittag?!}
\emmax{Ja, ein Mensch braucht schließlich seine 17 Stunden Schlaf\dots}
\miladx{\direct{entsetzt} 17 Stunden?! Das ist doch Quatsch! Ich will doch was erleben!}
\wilmax{Pfui Teufel!}
\emmax{Ach, stress' dich doch mal nicht so. Chill' doch einfach und genieß es! Zum was Erleben hast du auch später noch Zeit\dots}
\end{drama}

\StageDir{Hintergrund freeze, Spot auf \milamh auf der Bühnenkante.}

\begin{drama}
\milamx{Irgendwie wurde es mir echt unbehaglich, und ich habe nach einem Ausweg gesucht. Da wurde es langsam immer dunkler, und ein \sturmd näherte sich der Küste\dots}
\end{drama}

\StageDir{Irgendwie den \sturmd verdeutlichen (vielleicht ein stärkeres Meeresrauschen oder ein leises Gewitter im Hintergrund einspielen). Licht wieder an.}

\begin{drama}
\miladx{Ähm Leute, sollen wir nicht langsam mal reingehen? Das sieht nach einem Gewitter aus! Sowas ist am Strand doch echt gefährlich!}
\charlottex{Was meinst du? \direct{schaut auf} Oh, da sind aber viele Wolken!}
\wilhelminax{So eine Mistrotzkacke!}
\sophiax{Ach, so ein Quatsch, es regnet vielleicht gleich, na und? Ist doch kein Grund, sich jetzt so aufzuregen!}
\albertx{Meine wissenschaftliche Intuition sagt mir aber, dass das nach mehr als nur Regen aussieht!}
\cosmax{Kommt, dann gehen wir halt rein. Bevor sich noch einer verletzt\dots}
\cleox{Ach, chillt doch mal euer Life! Wir werden schon nicht draufgehen!}
\emmax{Ihr solltet wirklich mal einen Gang zurückschalten.}
\end{drama}

\StageDir{\miladh, \alberth, \wilmah, \wilhelminah und \cosmah verlassen die Bühne (gehen rein). Spot auf \milamh auf der Bühnenkante.}

\begin{drama}
\milamx{Irgendwie habe ich nach und nach gemerkt, dass diese Welt auch nicht das Richtige war. Den ganzen Tag nichts tun? Dann verpasse ich doch alles und erlebe nichts! Und mit der Zeit wurde es auch echt langweilig. Andauernd haben sie irgendwelche Dinge gesehen, die ich nicht gesehen habe, so wie \pikachuhh. Ich wollte mehr unternehmen und erleben, und nicht so langweilig rumhängen! Da waren \alberthh, \wilhelminahh und \wilmahh auch keine große Hilfe\dots}
\end{drama}

\StageDir{Licht aus.}
