\section{Liebes Tagebuch\dots}
\DisplayPersons
\DisplayRequisites
\ort{Am Strand in der entspannten Welt}

\StageDir{Spot auf \milamh am Bühnenrand.}

\begin{drama}
\milamx{Liebes Tagebuch, heute war wieder ein ganz verrückter Tag. Irgendwie sind wir durch eine Traumreise in eine andere Welt gekommen! Ein schönes Gefühl, weil Albert erst nicht glauben wollte, dass das funktioniert. Diesmal hatte ich recht! Dann habe ich diese coolen, gemütlichen Menschen getroffen. Die waren echt nett, total entspannt! Irgendwie beeindruckend. Aber ich weiß nicht\dots\ Nach einiger Zeit ist mir bewusst geworden, \textbf{WIE} gleichgültig sie waren. Und irgendwie wirkten die ein bisschen ZU entspannt. Und wollten mir die ganze Zeit etwas andrehen, ich weiß gar nicht, was das war. Aber Geheuer war es mir nicht\dots}

\end{drama}
