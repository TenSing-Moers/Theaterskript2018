\section{Der Schein trügt\dots}
\DisplayPersons
\DisplayRequisites
\ort{Auf der Hauptstraße in der Wutwelt}

\StageDir{Spot auf \milamh auf der Bühnenkante.}

\milamx{Liebes Tagebuch, eigentlich war die neue Welt genau das, was ich mir erträumt hatte. Nicht mehr das ständige Gefrage von Mama. Ich konnte endlich machen, was ich wollte, und niemand war direkt eine Spaßbremse, aus Angst vor irgendwelchen nicht vorhandenen Gefahren. Eigentlich perfekt! Naja, bis auf das \alberthh so nervig war. Aber ihm habe ich es ja eigentlich auch zu verdanken, dass ich überhaupt erst hier bin. Doch als ich von der Seitenstraße auf die Hauptstraße abbog, bekam meine Illusion Risse\dots}
\miladx{\direct{Kommt auf die Bühne} Wow, hier ist ja was los!}

\StageDir{Man sieht viele Statisten auf der Bühne, die sich anrempeln, schubsen, in der Ecke ist eine Schlägerei, etc. Es gibt einen \verkaufsstandd, alle tragen Hoodies, es liegt \muelld herum, richtiger Gangster-Style. \charlotteh und \sophiah sind auch irgendwo dabei.}

\miladx{\direct{schaut sich verwirrt um, spricht \charlotteh an} Entschuldigung, darf ich sie was fragen?}
\charlottex{\direct{aggressiv} Verschwinde, du dumme Göre, oder es wird dir noch Leid tun!}
\miladx{\direct{verwirrt, erschrocken, zögerlich} Oh\dots\ Okay, vielleicht frage ich besser jemand anderen\dots}
\miladx{\direct{geht zu \sophiah} Entschuldigen Sie bitte\dots}
\sophiax{\direct{aggressiv} Willst du Stress?! \direct{Droht ihr mit der Faust}}

\StageDir{Spot auf \milamh auf der Bühnenkante}

\milamx{Niemand konnte mir helfen. Alle haben mir nur blöde Antworten gegeben, dabei wusste ich nicht mal, was ich falsch gemacht habe. Je länger ich dort war, desto mehr fiel mir die schlechte Stimmung auf. Alle scheinen sich zu hassen. Weil ich langsam Durst hatte, bin ich zu einem Stand gegangen, um mir Wasser zu kaufen.}

\StageDir{\miladh geht im Hintergrund zu dem \verkaufsstandd. \emmah und \\\cleoh stehen irgendwo daneben. Der Verkäufer (\aaronh) schaut grimmig, sie wird erst noch beiseite geschubst von jemandem, der sich vorbeidrängelt.}

\miladx{\direct{freundlich} Guten Tag, ich hätte gerne eine Flasche Wasser.}
\aaronx{\direct{antwortet nicht, sondern streckt nur die Hand aus.}}
\miladx{\direct{verunsichert} Ehm\dots\ Wie viel kostet denn eine Flasche Wasser?}
\aaronx{\direct{unfreundlich} Guck doch selbst. Es ist immernoch derselbe Preis, 3 Wutgroschen. Mach doch mal die Augen auf, du dummes Mädel!}
\miladx{\direct{eingeschüchtert} Ich\dots\ habe hier drei Mark. Ist das richtig?}
\aaronx{\direct{wütend} Hast du mir nicht zugehört, du dämliche Göre?! Was soll das denn, willst du mich verarschen? Hier bezahlt man mit Wutgroschen, und nicht mit so einem Schrott! Du willst mich wohl bestehlen?!}
\emmax{\direct{hellhörig} Was?! Ein Dieb? Den müssen wir bestrafen!!}
\miladx{\direct{rechtfertigend} Was? Nein! Ich bin doch kein Dieb! Ich wollte doch nur\dots}
\aaronx{Haltet den Dieb!}
\cleox{\direct{zu \emmah} Schnell, die kriegen wir!}

\StageDir{\miladh wird von der Bühne gejagt und dabei beleidigt. Spot auf \milamh an der Bühnenkante.}

\milamx{Da seht ihr, was passiert ist. Ich bin so behandelt worden, wie der Dieb am Anfang, dabei habe ich gar nichts falsch gemacht. Es geht den Leuten nicht darum, Verbrechen zu bekämpfen, hier hasst einfach jeder jeden! Ich will einfach nur noch weg hier. Irgendwohin, wo sich nicht alle hassen, sondern sich umeinander kümmern! Aber, wo ist Einstein?}

\StageDir{Licht aus.}
